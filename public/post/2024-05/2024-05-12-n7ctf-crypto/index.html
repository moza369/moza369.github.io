`def encrypt(message, key):
    encrypted_message = ""
    key_index = 0
    for char in message:

        encrypted_char = chr(ord(char) + key_value)
        encrypted_message += encrypted_char
        key_index = (key_index + 1) % len(key)
    return encrypted_message
